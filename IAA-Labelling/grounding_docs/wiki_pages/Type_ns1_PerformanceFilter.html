Schema Definition

The GraphQL schema definition for this type: 

input PerformanceFilter {
  # Filter the result by the given Pulsar Job(s)
  jobs: [String]
  # Time period expressed as a difference from the current
# timestamp. Period can be used alone, without specifying start or end;
# i.e. period=1h means "the last one hour". It also can be used with
# either start or end, but not both. I.e. period=1h&start=1707591600
# means "calculate end time by adding one hour to start". In the current
# implementation, if you specify something like
# period=1h&end=1707591600, the end date is ignored and the time range
# is treated as only period=1h.
  period: String
  # (absolute or relative) for the beginning of the time range (ISO8601)
  startTime: StartTimeFrame
  # (absolute or relative) for the end of the time range (ISO8601)
  endTime: EndTimeFrame

  # Geographic region. It can be specified as a country code or
# country with subdivision. Also, 6 major regions can be used: NA (North
# America), SA (South Africa), AF (Africa), EU (Europe) and OC
# (Oceania). To differentiate between Country/Subdivision and Region,
# the Region must have a R_ prefix.
  area: String
  # Autonomous System Number (ASN) by which to filter results.
  asn: Int

  # filter by prefix like "R_"
  regionParentPrefix: String
}
