Schema Definition

The GraphQL schema definition for this type: 

# Identifies metrics available for reporting.
enum ReportMetric {
  # Fairshare Cost (Adjusted Amortized) - This is calculated by dividing the sum of the used resources for the container by the total used resources of all containers running on the same hardware, and then multiplying it by the cost of the hardware
  adjusted_amortized_fairshare_cost
  # Idle Cost (Adjusted Amortized) - The idle cost refers to the cost incurred by resources that are not used by any workload for a given time range
  adjusted_amortized_idle_cost
  # Utilized Cost (Adjusted Amortized) - The utilized cost refers to the cost incurred by the resources utilized by a workload for a given time range (can be further broken down or grouped by cluster, namespace, labels)
  adjusted_amortized_utilized_cost
  # Cost (Adjusted) - Built from the cost (total) metric, this metric adjusts the figures to allow for any custom pricing rules. It only appears for Cloudability customers who have turned on the custom pricing module.
  adjusted_cost
  # Fairshare Cost (Adjusted) - This is calculated by dividing the sum of the used resources for the container by the total used resources of all containers running on the same hardware, and then multiplying it by the cost of the hardware
  adjusted_fairshare_cost
  # Idle Cost (Adjusted) - The idle cost refers to the cost incurred by resources that are not used by any workload for a given time range
  adjusted_idle_cost
  # Utilized Cost (Adjusted) - The utilized cost refers to the cost incurred by the resources utilized by a workload for a given time range (can be further broken down or grouped by cluster, namespace, labels)
  adjusted_utilized_cost
  # Rate (Blended) - For Consolidated Billing accounts, the effective rate for the line item calculated as an average of the cost of identical instances operating in that hour in the same Availability Zone.
  blended_rate
  # Cost (Commited)
  business_metric1
  # Someone's Test
  business_metric10
  # Bytes Transferred Passthrough
  business_metric11
  # Cost of East
  business_metric13
  # CBR Savings Plans Savings
  business_metric14
  # test it
  business_metric15
  # Test_UsageFam
  business_metric2
  # test_EC2_COST
  business_metric3
  # test_powerup
  business_metric4
  # GST
  business_metric5
  # test demo
  business_metric6
  # Savings testak
  business_metric7
  # Account Count
  business_metric8
  # Surcharge
  business_metric9
  # GiB Hours - A gigabyte-hour refers to the number of gigabytes stored on a cloud platform in a particular hourly period.
  byte_hours
  # Data Transfer (GiB) - The amount of data transferred in a particular time period.
  bytes_transferred
  # Estimated Carbon Emissions (MTCO2e) - Estimated Carbon emissions by a resource in metric tons of carbon dioxide equivalent - MTCO2e
  carbon_emissions
  # Cost Adjustment - The amount by which the cost was adjusted.
  cost_adjusted
  # GiB Months - References byte-hours, which corresponds to the number of bytes stored on an AWS, Azure, GCP or OCI storage service in a particular hourly period.
  gb_months
  # Cost (Total Blended) - This cost metric is specific to AWS and corresponds to the lineItem/BlendedCost column of the CUR file. This metric “blends” the impact of reserved instances and savings plans.
  invoiced_cost
  # I/O Requests - Describes the number of data requests from a database service.
  io_requests
  # Power Consumed (kWh) - Power consumed by a resource in Kilo Watt Hours - kWh
  kwh_consumption
  # On Demand Hours - The amount of on demand computing hours used during the reporting period.
  on_demand_hours
  # Cost (List) - This metric represents cost items as if they were charged at the public on-demand rate. This means the impact of any commitments, spot instance pricing, and custom pricing are removed.
  public_on_demand_cost
  # Requests - The number of computing resources used in the units specified by each service. For example, your service may determine pricing by the amount of storage, number of requests, or hours of operation.
  requests
  # Reserved Hours - The amount of reserved computing hours used during the reporting period.
  reserved_hours
  # Reserved Coverage Rate - The percentage of total hours used during the period where an RI was applied.
  reserved_utilization_rate
  # Resource Count - The count of ?unique? resource ids in the period specified.
  resource_identifier_count
  # Cost (Adjusted Amortized) - Built from the cost (amortized) metric, this metric adjusts the figures to allow for any custom pricing rules. It only appears for Cloudability customers who have turned on the custom pricing module.
  total_adjusted_amortized_cost
  # Cost (Amortized) - This is an “accrual” cost metric, representing the consumed cost during any period. This metric is handy for ensuring commitments are allocated to the time and place they are consumed.
  total_amortized_cost
  # Cost (Total) - This is the default cost metric throughout Cloudability. It is a “cash” metric and simply represents the invoiced amount associated with any cost item as reported by the cloud vendor.
  unblended_cost
  # Rate (Unblended) - The cost per hour for a line item. This is summarized across the hours of a day.
  unblended_rate
  # Usage Hours - The number of computing resources used in the units specified by each service. For example, your service may determine pricing by the amount of storage, number of requests, or hours of operation.
  usage_hours
  # Usage Quantity - Generic usage quantity. Units will vary on dimensions and filters.
  usage_quantity
  # vCPU Hours - Normalized CPU hours are calculated by multiplying the CPU hours used by the number of vCPUs used.
  vcpu_hours
}
